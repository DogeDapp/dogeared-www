<form class="form" role="form" method="POST" id="fetch">

      <div class="form-group sr">
      <label for="url"></label>
	<input type="text" name="url" id="url" class="form-control" />
	</div>
	<button class="btn btn-primary pull-right">Dogear these words</button>
</form>

</div>

<script type="text/javascript">
{literal}

$(document).ready(function(){

	$("#fetch").submit(function(){

		var url = $("#url");
		url = url.val();

		var on_success = function(rsp){

			dogeared_feedback("Okay! That document is ready to be dogeared.");

			var document = rsp['document'];
			dogeared_cache_documents_store(document);
		};


		var method = "dogeared.readinglists.addDocument";
		var args = { 'url': url };

		dogeared_api_call(method, args, on_success);
		return false;
	});
});

{/literal}
</script>
