<script type="text/javascript">
{literal}

$(document).ready(function(){

	$("#fetch").submit(function(){

		var url = $("#url");
		url = url.val();

		var onsuccess = function(rsp){
			var document = rsp['document'];
			dogeared_cache_documents_store(document);
		};

		var method = "dogeared.readinglists.addDocument";
		var args = { 'url': url };

		// console.log(args);
		// return false;

		dogeared_api_call(method, args, onsuccess);
		return false;
	});
});

</script>
